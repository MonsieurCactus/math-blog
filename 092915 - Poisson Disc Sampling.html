def circ(z, r):
    dt = 0.01
    t  = np.arange(0,1+dt,dt)
    return z + r*np.exp(2j*np.pi*t)

C = []

alphabet = '0123456789ABCDE'

while len(C) < 500:
    r     = np.sqrt(np.random.random())
    theta = 2*np.pi*np.random.random()
    ctr   = r*np.exp(2j*np.pi*theta)
    
    D = [ np.abs(c[0] - ctr) - c[1] for c in C ] + [1-r]
    
    if np.all([(x > 0) for x in D]):
        dist = min(D)

        R = np.sqrt(np.random.random())*np.abs(dist)
        z = circ( ctr , R )

        C += [( ctr , R )]
